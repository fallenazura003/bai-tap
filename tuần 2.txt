1.#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int n;
    cin >> n;
    if(n==0||n==1)
        cout << "no";
    else
    {
        int count=0;
        for(int i=2;i<=sqrt(n);i++)
        {
            if(n%i==0)
                count ++;
        }
        if(count)
            cout << "no";
        else
            cout << "yes";
    }
}
2.#include <iostream>

using namespace std;

int main()
{
    int n;
    cin >> n;
    for(int i=1;i<=n;i++){
        for(int k=n-i;k<n-1;k++)
            cout << ' ';
        for(int j=n+1-i;j>0;j--)
            {
                cout<<"*";
            }
        cout << endl;
    }
}

{

            cout << ' ';
