#include<iostream>
#include<cmath>

using namespace std;
bool checkNT(int a)
{
	if (a < 2)
		return false;
	else
	{
		int n = 0;
		for (int i = 2; i <= sqrt(a); i++)
		{
			if (a % i == 0)
			{
				n++;
			}
		}
		if (n == 0)
			return true;
		return false;
	}
}

int main(int argc, char* args[])
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		if (checkNT(i))
			cout << i << ' ';
	}
	return 0;
}

#include<iostream>

using namespace std;

int UCLN(int a, int b)
{
	if (a == 0 || b == 0){
        return a + b;
    }
    while (a != b){
        if (a > b){
            a -= b; 
        }else{
            b -= a;
        }
    }
    return a;
}

int main(int argc, char* args[])
{
	int a,b;
	cin >> a >> b;
	UCLN(a, b);
	return 0;
}

#include<iostream>
using namespace std;


void vetamgiac(int a)
{
	int m, n;
	n = 1;
	while (n <= a)
	{
		m = (a - n) / 2;
		for (int i = 0; i < m; i++)
		{
			cout << ' ';
		}
		for (int i = 0; i < n; i++)
		{
			cout << '*';
		}
		cout <<endl;
		n = n + 2;
	}
}
int main(int argc, char* args[])
{
	int  a;
	cin >> a;
	vetamgiac(a);
	return 0;
}
