3.#include<iostream>
using namespace std;

struct point
{
	int x,y;
	point(){};
	point(int _x, int _y): x(_x) , y(_y){
	};
		
};

point mid_point(const point& a,const point& b)
{
	point mid;
	mid.x = (a.x + b.x)/2;
	mid.y = (a.y + b.y)/2;
	return mid;
}
int main()
{
	point a(9,9);
	point b(23,10);
	point c;
	c= mid_point(a,b);
	cout << '(' << c.x << ',' << c.y << ')';
	return 0;
}

struct String{
    int n;
    char* str;
    String(const char* s){
        n=strlen(s);
        str=new char[n];
        strncpy(str,s,n);
    }
    ~String(){
        delete [] str;
    }
    void print(){cout << str << endl;}
     void append(const char* ss){
        int l=n+strlen(ss);
        char* sstr=new char[l];
        strncpy(sstr,str,n);
        for(int i=n;i<l;i++){
            sstr[i]=ss[i-n];
        }
        sstr[l]='\0';
        delete str;
        str=sstr;
    }
};
int main (){
    String s("Hi");
    s.append(" there");
    cout << s;
